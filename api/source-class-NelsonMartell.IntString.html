 78         /**
 79          * Convert the object to an instance of ``IntString``.
 80          *
 81          * @param string|IntString $obj Object to convert to ``IntString``.
 82          *
 83          * @return IntString
 84          * @throws InvalidArgumentException if object is not a string or format is invalid.
 85          */
 86         public static function parse($obj)
 87         {
 88             if ($obj instanceof IntString) {
 89                 return $obj;
 90             }
 91 
 92             if (is_integer($obj)) {
 93                 return new VersionComponent($obj);
 94             }
 95 
 96             try {
 97                 $intValue = (integer) String::ensureIsString($obj);
 98             } catch (InvalidArgumentException $e) {
 99                 $args = [
100                     'position' => '1st',
101                     'expected' => 'string" or "integer',
102                     'actual'   => typeof($obj),
103                 ];
104 
105                 $msg = nml_msg('Invalid argument type.');
106                 $msg .= nml_msg(
107                     ' {position} parameter must to be an instance of "{expected}"; "{actual}" given.',
108                     $args
109                 );
110 
111                 throw new InvalidArgumentException($msg, 1, $e);
112             }
113 
114             $stringValue = ltrim($obj, "$intValue");
115 
116             // Validate that 0 (zero) is not interpreted as '' (empty string)
117             if ($stringValue === $obj) {
118                 $msg = nml_msg('Invalid argument value.');
119                 $msg .= nml_msg(' "{0}" (string) must to start with an integer.', $obj);
120 
121                 throw new InvalidArgumentException($msg);
122             }
123 
124             return new IntString($intValue, $stringValue);
125         }
