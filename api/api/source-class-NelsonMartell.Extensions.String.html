 31     /**
 32      * Replaces format elements in a string with the string representation of an
 33      * object matching the list of arguments specified. You can give as many
 34      * params as you need, or an array with values.
 35      *
 36      * ##Usage
 37      * Using numbers as placeholders (encloses between `{` and `}`), you can get
 38      * the matching string representation of each object given. Use `{0}` for
 39      * the fist object, `{1}` for the second, and so on.
 40      * Example:
 41      * `String::format('{0} is {1} years old, and have {2} cats.', 'Bob', 65, 101);`
 42      * Returns: 'Bob is 65 years old, and have 101 cats.'
 43      *
 44      * You can also use an array to give objects values.
 45      * Example: `String::Format('{0} is {1} years old.', ['Bob', 65, 101]);`
 46      * Returns: 'Bob is 65 years old, and have 101 cats.'
 47      *
 48      * If give an key => value array, each key stands for a placeholder variable
 49      * name to be replaced with value key. In this case, order of keys do not
 50      * matter.
 51      * Example:
 52      * `$arg0 = ['name' => 'Bob', 'n' => 101, 'age' => 65];`
 53      * `$format = '{name} is {age} years old, and have {n} cats.';`
 54      * `String::Format($format, $arg0);`
 55      * Returns: 'Bob is 65 years old, and have 101 cats.'
 56      *
 57      * @param string      $format A string containing variable placeholders.
 58      * @param array|mixed $args   Object(s) to be replaced into $format
 59      *   placeholders.
 60      *
 61      * @return string
 62      * @todo   Implement, for php 5.6+:
 63      *   php.net/functions.arguments.html#functions.variable-arg-list.new
 64      * @todo   Implement formatting, like IFormatProvider or something like that.
 65      * @author Nelson Martell <nelson6e65-dev@yahoo.es>
 66      */
 67     public static function format($format, $args)
 68     {
 69         static $options = [
 70             'before'  => '{',
 71             'after'   => '}',
 72         ];
 73 
 74         $data = func_num_args() === 2 ? (array) $args : array_slice(func_get_args(), 1);
 75 
 76         return static::insert($format, $data, $options);
 77     }

 97     /**
 98      * Ensures that object given is an string. Else, thows an exception
 99      *

102      * @return string Same object given, but ensured that is an string.
103      * @throws InvalidArgumentException if object is not an `string`.

105     public static function ensureIsString($obj)
106     {
107         if (!is_string(static::ensureIsNotNull($obj))) {
108             $msg = nml_msg('Provided object must to be an string; "{0}" given.', typeof($obj));

152     /**
153      * Ensures that an string follows the PHP variables naming convention.
154      *
155      * @param string $string String to be ensured.
156      *
157      * @return string
158      * @throws InvalidArgumentException if object is not an `string` or do not
159      *   follows the PHP variables naming convention.
160      */
161     public static function ensureIsValidVarName($string)
162     {
163         $pattern = '/^[a-zA-Z_\x7f-\xff][a-zA-Z0-9_\x7f-\xff]*/';
164 
165         if (!preg_match($pattern, static::ensureIsString($string))) {
166             $msg = nml_msg('Provided string do not follows PHP variables naming convention: "{0}".', $string);
167             throw new InvalidArgumentException($msg);
168         }
169 
170         return $string;
