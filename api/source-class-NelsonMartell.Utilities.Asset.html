270         /**
271          * Obtiene la ruta del directorio de la versión especificada. Si no se
272          * especifica, se devuelve la versión más reciente.
273          *
274          * @param string|Version $version Versión a obtener. También puede
275          *   tomar los valores 'newest' u 'oldest' para representar a la versión
276          *   más nueva o más vieja, respectivamente.
277          *
278          * @return string Ruta del directorio de la versión especificada.
279          * */
280         public function getDirectoryPath($version = self::NEWEST)
281         {
282             $c = count($this->Versions);
283 
284             if ($c == 0) {
285                 throw new LogicException(nml_msg('This Asset has not versions.'));
286             }
287             $v = $version;
288 
289             if ($version == self::OLDEST or $version == self::NEWEST) {
290                 $v = $this->Versions[0];
291 
292                 if ($c > 1) {
293                     usort($this->versions, array("NelsonMartell\\Version", "Compare"));
294 
295                     $v = $this->Versions[0];
296 
297                     if ($version == self::NEWEST) {
298                         $v = $this->Versions[$c - 1];
299                     }
300                 }
301             } else {
302                 try {
303                     $v = Version::parse($version);
304                 } catch (InvalidArgumentException $e) {
305                     $args = [
306                         'name'     => 'version',
307                         'pos'      => 0,
308                         'expected' => typeof(new Version(1, 0))->Name,
309                         'actual'   => typeof($version),
310                     ];
311 
312                     $msg = nml_msg('Invalid argument type.');
313                     $msg .= nml_msg(
314                         ' "{name}" (position {pos}) must to be an instance of "{expected}" (or compatible);'.
315                         ' "{actual}" given.',
316                         $args
317                     );
318 
319                     throw new InvalidArgumentException($msg);
320                 }
321 
322                 if (array_search($v, $this->Versions) === false) {
323                     $msg = nml_msg('Invalid argument value.');
324                     $msg .= nml_msg(' Asset has not the version "{version}".', ['version' => $v]);
325 
326                     throw new InvalidArgumentException($msg);
327                 }
328             }
329 
330             return sprintf('%s%s/', $this->RootDirectory, $v);
331         }
